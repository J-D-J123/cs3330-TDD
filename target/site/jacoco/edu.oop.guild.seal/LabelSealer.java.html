package edu.oop.guild.seal;

import edu.oop.guild.model.DeliveryRequest;
import edu.oop.guild.model.RealmType;

import java.util.Map;
import java.util.Objects;

public class LabelSealer {
    private static final Map<RealmType, String> EMBLEMS = Map.of(
            RealmType.SKY, "☁",
            RealmType.UNDERGROUND, "⛏"
    );

    public String seal(DeliveryRequest request) {
        Objects.requireNonNull(request);
        String emblem = EMBLEMS.get(request.getDestinationRealm());
        return emblem + " " + request.getPackageType().label()
                + " to " + request.getDestinationRealm().displayName()
                + " " + emblem;
    }
}
