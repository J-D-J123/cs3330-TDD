package edu.oop.guild.factory;

import edu.oop.guild.creature.Creature;
import edu.oop.guild.model.DeliveryPlan;
import edu.oop.guild.model.DeliveryRequest;
import edu.oop.guild.seal.LabelSealer;
import edu.oop.guild.strategy.DeliveryCostStrategy;

import java.util.Objects;

public class DeliveryPlanFactory {
    private final Creature courier;
    private final DeliveryCostStrategy strategy;
    private final LabelSealer sealer;

    public DeliveryPlanFactory(Creature courier, DeliveryCostStrategy strategy, LabelSealer sealer) {
        this.courier = Objects.requireNonNull(courier);
        this.strategy = Objects.requireNonNull(strategy);
        this.sealer = Objects.requireNonNull(sealer);
    }

    public DeliveryPlan create(DeliveryRequest request) {
        Objects.requireNonNull(request);
        String label = sealer.seal(request);
        int price = strategy.estimateCoins(request);
        return new DeliveryPlan(request, courier, label, price);
    }
}
